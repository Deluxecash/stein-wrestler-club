        function openPopup({ title, message, type = 'alert', label = 'PASSWORT BESTÄTIGEN' }) {
            return new Promise((resolve) => {
                const modal = document.getElementById('custom-modal');
                document.getElementById('modal-title').textContent = title;
                document.getElementById('modal-message').textContent = message;
                
                const promptArea = document.getElementById('modal-prompt-area');
                const promptInput = document.getElementById('modal-prompt-input');
                document.getElementById('modal-prompt-label').textContent = label;
                promptInput.value = '';

                const cancelBtn = document.getElementById('modal-btn-cancel');
                const confirmBtn = document.getElementById('modal-btn-confirm');

                if (type === 'alert') {
                    cancelBtn.classList.add('hidden');
                    promptArea.classList.add('hidden');
                } else if (type === 'confirm') {
                    cancelBtn.classList.remove('hidden');
                    promptArea.classList.add('hidden');
                } else if (type === 'promptPassword') {
                    cancelBtn.classList.remove('hidden');
                    promptArea.classList.remove('hidden');
                }

                modal.classList.remove('hidden');

                confirmBtn.onclick = () => { modal.classList.add('hidden'); resolve({ success: true, value: promptInput.value }); };
                cancelBtn.onclick = () => { modal.classList.add('hidden'); resolve({ success: false, value: null }); };
            });
        }

                    card.className = 'role-card is-locked';
                    card.innerHTML = `
                        <div class="role-card-header">
                            <div class="role-card-icon" style="filter: grayscale(1); opacity: 0.3;">${role.icon}</div>
                            <h3>${role.name}</h3>
                        </div>
                        <div class="info-text"><span>GESPERRT</span></div>`;
                    card.addEventListener('click', () => openLockedScreen(role));
                } else {
                    card.className = 'role-card';
                    let prog = currentUser.progress[role.id] || 0;
                    let perc = Math.min(100, Math.round((prog / role.max) * 100));
                    const currentRank = getRank(prog, role.max);

                    card.innerHTML = `
                        <div class="role-card-header">
                            <div class="role-card-icon" style="box-shadow: 0 0 10px ${role.color}; border-color:${role.color}; color:${role.color}">${role.icon}</div>
                            <h3>${role.name}</h3>
                        </div>
                        <div class="info-text">
                            <span style="display:flex; align-items:center; gap:4px; color:${currentRank.color}">
                                ${currentRank.icon} ${currentRank.name.toUpperCase()} (${perc}%)
                            </span>
                            <span>➔</span>
                        </div>`;
                    card.addEventListener('click', () => openMastery(role));
                }
                container.appendChild(card);
            });
        }
        document.getElementById('search-roles').addEventListener('input', (e) => renderRoles(e.target.value));

        function renderLeaderboard() {
            const tbody = document.getElementById('leaderboard-rows');
            tbody.innerHTML = '';
            const users = JSON.parse(localStorage.getItem('steinUsers')) || {};
            const mode = document.getElementById('leaderboard-type-select').value;
            const scoreHeader = document.getElementById('leaderboard-score-header');
            let leaderboardData = [];

            if (mode === 'global') {
                scoreHeader.textContent = 'Gesamt-Mastery';
                leaderboardData = Object.keys(users).map(username => {
                    const progressObj = users[username].progress || {};
                    let sumOfPercentages = 0;
                    rolesData.forEach(role => {
                        const prog = progressObj[role.id] || 0;
                        sumOfPercentages += Math.min(100, (prog / role.max) * 100);
                    });
                    const totalPercent = Math.round(sumOfPercentages / rolesData.length);
                    return { username, score: totalPercent, display: `${totalPercent}%`, maxLimit: 100 };
                });
            } else {
                const activeRole = rolesData.find(r => r.id === mode);
                scoreHeader.innerHTML = activeRole ? `<div id="leaderboard-score-header-content">${activeRole.icon} <span>${activeRole.name.toUpperCase()}</span></div>` : 'Fortschritt';
                leaderboardData = Object.keys(users).map(username => {
                    const progressObj = users[username].progress || {};
                    const prog = progressObj[mode] || 0;
                    const max = activeRole ? activeRole.max : 10000;
                    return { username, score: prog, display: `${prog} / ${max}`, maxLimit: max };
                });
            }

            leaderboardData.sort((a, b) => b.score - a.score);
            const isCurrentUserAdmin = currentUser && currentUser.username === "Deluxecash";

            leaderboardData.forEach((player, index) => {
                const tr = document.createElement('tr');
                const rank = index + 1;
                let rankClass = rank === 1 ? 'rank-1' : rank === 2 ? 'rank-2' : rank === 3 ? 'rank-3' : 'rank-other';
                const playerRankInfo = getRank(player.score, player.maxLimit);

                let adminActionButtons = '';
                if (isCurrentUserAdmin && player.username !== "Deluxecash") {
                    adminActionButtons = `
                        <button class="admin-remove-btn" onclick="deleteUserProgress('${player.username}')">(entf.)</button>
                        <button class="admin-ban-btn" onclick="banUserAccount('${player.username}')">(ban.)</button>
                    `;
                }

                tr.innerHTML = `
                    <td><span class="rank-badge ${rankClass}">${rank}</span></td>
                    <td>
                        <span style="${player.username === currentUser.username ? 'color: var(--accent-orange); font-weight: 700;' : ''}">${player.username}</span>${player.username === "Deluxecash" ? '<span class="admin-badge">[ADMIN]</span>' : ''}
                        <span class="player-rank-tag" style="color: ${playerRankInfo.color}; border: 1px solid ${playerRankInfo.color}44;">
                            ${playerRankInfo.icon} ${playerRankInfo.name}
                        </span>
                        ${adminActionButtons}
                    </td>
                    <td style="text-align: right; font-family: var(--font-gaming); color: var(--accent-orange); font-weight: 700;">${player.display}</td>
                `;
                tbody.appendChild(tr);
            });
        }
        document.getElementById('leaderboard-type-select').addEventListener('change', renderLeaderboard);

        window.deleteUserProgress = async function(targetUser) {
            if((await openPopup({ title: 'ENTFERNEN?', message: `Fortschritt von ${targetUser} löschen?`, type: 'confirm' })).success) {
                let users = JSON.parse(localStorage.getItem('steinUsers')) || {};
                if (users[targetUser]) { users[targetUser].progress = {}; localStorage.setItem('steinUsers', JSON.stringify(users)); renderLeaderboard(); }
            }
        };

        window.banUserAccount = async function(targetUser) {
            if((await openPopup({ title: 'BAN?', message: `Account von ${targetUser} löschen?`, type: 'confirm' })).success) {
                let users = JSON.parse(localStorage.getItem('steinUsers')) || {};
                if (users[targetUser]) { delete users[targetUser]; localStorage.setItem('steinUsers', JSON.stringify(users)); renderLeaderboard(); }
            }
        };

        function openLockedScreen(role) {
            showView('locked');
            document.getElementById('locked-char-name').innerHTML = `${role.icon} <span>${role.name}</span>`;
            document.getElementById('locked-condition-text').textContent = role.unlockCond || "Gesperrt.";
        }

        function openMastery(role) {
            currentRoleView = role;
            showView('mastery');
            document.getElementById('mastery-title').innerHTML = `${role.icon} <span>${role.name}</span>`;
            document.getElementById('mastery-title').style.setProperty('--mastery-color', role.color);
            document.documentElement.style.setProperty('--mastery-color', role.color);
            document.getElementById('mastery-desc').textContent = role.desc;
            document.getElementById('mastery-task').textContent = role.task;
            
            // HIER WAR DER FEHLER: classList korrigiert!
            document.getElementById('mastery-standard-content').classList.remove('hidden');
            updateMasteryUI();
        }

        function updateMasteryUI() {
            let prog = currentUser.progress[currentRoleView.id] || 0;
            const max = currentRoleView.max;
            document.getElementById('mastery-numbers').textContent = `${prog} / ${max}`;
            const perc = Math.min(100, Math.round((prog / max) * 100));
            document.getElementById('mastery-bar').style.width = `${perc}%`;
            document.getElementById('mastery-percent').textContent = `${perc}%`;
            
            const slider = document.getElementById('mastery-slider');
            slider.max = max - prog; 
            slider.value = 0;
            document.getElementById('mastery-slider-val').textContent = '+0';
        }
        
        document.getElementById('mastery-slider').addEventListener('input', (e) => {
            document.getElementById('mastery-slider-val').textContent = `+${e.target.value}`;
        });

        document.getElementById('btn-submit-mastery').addEventListener('click', () => {
            const val = parseInt(document.getElementById('mastery-slider').value);
            if (val > 0) {
                let currentProg = currentUser.progress[currentRoleView.id] || 0;
                let newProg = Math.min(currentRoleView.max, currentProg + val);
                currentUser.progress[currentRoleView.id] = newProg;
                saveUserData();
                updateMasteryUI();
            }
        });

        function saveUserData() {
            let users = JSON.parse(localStorage.getItem('steinUsers'));
            users[currentUser.username].progress = currentUser.progress;
            localStorage.setItem('steinUsers', JSON.stringify(users));
            localStorage.setItem('steinUser', JSON.stringify(currentUser));
        }

        document.getElementById('change-username-form').addEventListener('submit', async (e) => {
            e.preventDefault();
            const newName = document.getElementById('settings-new-username').value.trim();
            let users = JSON.parse(localStorage.getItem('steinUsers')) || {};
            if (users[newName]) { await openPopup({ title: 'Fehler', message: 'Name vergeben!' }); return; }
            users[newName] = users[currentUser.username];
            delete users[currentUser.username];
            currentUser.username = newName;
            localStorage.setItem('steinUsers', JSON.stringify(users));
            localStorage.setItem('steinUser', JSON.stringify(currentUser));
            document.getElementById('nav-username').textContent = newName;
            await openPopup({ title: 'Erfolg', message: 'Name geändert!' });
        });

        document.getElementById('change-pass-form').addEventListener('submit', async (e) => {
            e.preventDefault();
            const currInput = document.getElementById('settings-curr-pass').value;
            const newInput = document.getElementById('settings-new-pass').value;
            let users = JSON.parse(localStorage.getItem('steinUsers'));
            if (currInput !== users[currentUser.username].password) { await openPopup({ title: 'Fehler', message: 'Falsches Passwort!' }); return; }
            users[currentUser.username].password = newInput;
            localStorage.setItem('steinUsers', JSON.stringify(users));
            await openPopup({ title: 'Erfolg', message: 'Passwort aktualisiert!' });
        });

        document.getElementById('btn-reset-progress').addEventListener('click', async () => {
            if ((await openPopup({ title: 'Sicher?', message: 'Fortschritt nullen?', type: 'confirm' })).success) {
                currentUser.progress = {}; saveUserData(); renderRoles(); showView('roles');
            }
        });

        document.getElementById('btn-delete-acc').addEventListener('click', async () => {
            const promptRes = await openPopup({ title: 'LÖSCHEN?', message: 'Gib dein Passwort ein.', type: 'promptPassword' });
            if (promptRes.success) {
                let users = JSON.parse(localStorage.getItem('steinUsers'));
                if (promptRes.value === users[currentUser.username].password) {
                    delete users[currentUser.username]; localStorage.setItem('steinUsers', JSON.stringify(users));
                    localStorage.removeItem('steinUser'); currentUser = null;
                    document.getElementById('navbar').classList.add('hidden'); showView('login');
                }
            }
        });
